# samsungcloudplatformv2_invitation_decline (Resource)

Invitation Decline - Use a separate provider alias with target account credentials

## Example Usage

```terraform
provider "samsungcloudplatformv2" {
}

provider "samsungcloudplatformv2" {
  alias      = "target_account"
  access_key = var.target_access_key
  secret_key = var.target_secret_key
}

resource "samsungcloudplatformv2_invitation_decline" "decline" {
  provider = samsungcloudplatformv2.target_account
  id       = var.invitation_id
}

output "invitation" {
  description = "Invitation details"
  value       = samsungcloudplatformv2_invitation_decline.decline.invitation
}

variable "invitation_id" {
  description = "Invitation ID to decline"
  type        = string
  default     = ""
}

variable "target_access_key" {
  description = "Target account access key"
  type        = string
  sensitive   = true
  default     = ""
}

variable "target_secret_key" {
  description = "Target account secret key"
  type        = string
  sensitive   = true
  default     = ""
}
```
