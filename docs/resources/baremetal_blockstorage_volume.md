# samsungcloudplatformv2_baremetal_blockstorage_volume (Resource)

Block Storage(BM)

## Example Usage

```terraform
provider "samsungcloudplatformv2" {
  default_region = "kr-west1"
}

resource "samsungcloudplatformv2_baremetal_blockstorage_volume" "volume"{
  name = var.block_storage_name
  disk_type = var.disk_type
  size_gb = var.size_gb
  attachments = var.attachments
  qos = var.qos
  tags = var.tags
  timeouts {
    create = var.create_timeouts
    delete = var.delete_timeouts
  }
}


output "block_storage_output" {
  value= samsungcloudplatformv2_baremetal_blockstorage_volume.volume
  description = "description"
}


variable "block_storage_name" {
  type    = string
  default = "terraform-bs-01"
}

variable "disk_type" {
  type    = string
  default = "SSD"
}

variable "size_gb" {
  type    = number
  default = 10
}

variable "attachments" {
  type = list(object({
    object_type = string
    object_id   = string
  }))
  default = []
}

variable "qos" {
  type = object({
    iops       = number
    throughput = number
  })
  default = {
    iops       = 5000
    throughput = 250
  }
}

variable "tags" {
  type = map(string)
  default = {
    no_value = ""
    tf_key   = "tf_value"
  }
}

variable "create_timeouts" {
  type    = string
  default = "20m"
}

variable "delete_timeouts" {
  type    = string
  default = "20m"
}
```
