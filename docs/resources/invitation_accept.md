# samsungcloudplatformv2_invitation_accept (Resource)

Invitation Accept - Use a separate provider alias with target account credentials

## Example Usage

```terraform
provider "samsungcloudplatformv2" {
}

provider "samsungcloudplatformv2" {
  alias = "target_account"
  access_key = var.target_access_key
  secret_key = var.target_secret_key
}

resource "samsungcloudplatformv2_invitation_accept" "accept" {
  provider = samsungcloudplatformv2.target_account
  id       = var.invitation_id
}

output "master_account_email" {
  description = "Master Account Email"
  value       = samsungcloudplatformv2_invitation_accept.accept.master_account_email
}

output "success_id" {
  description = "Success ID object"
  value       = samsungcloudplatformv2_invitation_accept.accept.success_id
}

output "failed_id" {
  description = "Failed ID object"
  value       = samsungcloudplatformv2_invitation_accept.accept.failed_id
}

variable "invitation_id" {
  description = "Invitation ID to accept"
  type        = string
  default     = ""
}

variable "target_access_key" {
  description = "Target account access key"
  type        = string
  sensitive   = true
  default     = ""
}

variable "target_secret_key" {
  description = "Target account secret key"
  type        = string
  sensitive   = true
  default     = ""
}
```
